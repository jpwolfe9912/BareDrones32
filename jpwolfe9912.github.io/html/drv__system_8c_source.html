146uint32_t
147micros(void)
148{
149 register uint32_t oldCycle, cycle, timeMs;
150
151 do
152 {
153 timeMs = __LDREXW(&sysTickUptime);
154 cycle = DWT->CYCCNT;
155 oldCycle = sysTickCycleCounter;
156 }
157 while ( __STREXW( timeMs , &sysTickUptime ) );
158
159 return (timeMs * 1000) + (cycle - oldCycle) / usTicks;
160}

185void delayMicroseconds(uint32_t us)
186{
187 uint32_t elapsed = 0;
188 uint32_t lastCount = DWT->CYCCNT;
189
190 for (;;) {
191 register uint32_t current_count = DWT->CYCCNT;
192 uint32_t elapsed_us;
193
194 // measure the time elapsed since the last time we checked
195 elapsed += current_count - lastCount;
196 lastCount = current_count;
197
198 // convert to microseconds
199 elapsed_us = elapsed / usTicks;
200 if (elapsed_us >= us)
201 break;
202
203 // reduce the delay by the elapsed time
204 us -= elapsed_us;
205
206 // keep fractional microseconds for the next iteration
207 elapsed %= usTicks;
208 }
209}

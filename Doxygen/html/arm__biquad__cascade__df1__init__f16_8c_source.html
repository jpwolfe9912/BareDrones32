78void arm_biquad_cascade_df1_init_f16(
79 arm_biquad_casd_df1_inst_f16 * S,
80 uint8_t numStages,
81 const float16_t * pCoeffs,
82 float16_t * pState)
83{
84 /* Assign filter stages */
85 S->numStages = numStages;
86
87 /* Assign coefficient pointer */
88 S->pCoeffs = pCoeffs;
89
90 /* Clear state buffer and size is always 4 * numStages */
91 memset(pState, 0, (4U * (uint32_t) numStages) * sizeof(float16_t));
92
93 /* Assign state pointer */
94 S->pState = pState;
95}

99static void generateCoefsFastBiquadF16(float16_t b0, float16_t b1, float16_t b2, float16_t a1, float16_t a2,
100 arm_biquad_mod_coef_f16 * newCoef)
101{
102 float32_t coeffs[8][12] = {
103 {0, 0, 0, 0, 0, 0, 0, b0, b1, b2, a1, a2},
104 {0, 0, 0, 0, 0, 0, b0, b1, b2, 0, a2, 0},
105 {0, 0, 0, 0, 0, b0, b1, b2, 0, 0, 0, 0},
106 {0, 0, 0, 0, b0, b1, b2, 0, 0, 0, 0, 0},
107 {0, 0, 0, b0, b1, b2, 0, 0, 0, 0, 0, 0},
108 {0, 0, b0, b1, b2, 0, 0, 0, 0, 0, 0, 0},
109 {0, b0, b1, b2, 0, 0, 0, 0, 0, 0, 0, 0},
110 {b0, b1, b2, 0, 0, 0, 0, 0, 0, 0, 0, 0}
111 };
112
113 for (int i = 0; i < 12; i++)
114 {
115 coeffs[1][i] += (a1 * coeffs[0][i]);
116 coeffs[2][i] += (a1 * coeffs[1][i]) + (a2 * coeffs[0][i]);
117 coeffs[3][i] += (a1 * coeffs[2][i]) + (a2 * coeffs[1][i]);
118 coeffs[4][i] += (a1 * coeffs[3][i]) + (a2 * coeffs[2][i]);
119 coeffs[5][i] += (a1 * coeffs[4][i]) + (a2 * coeffs[3][i]);
120 coeffs[6][i] += (a1 * coeffs[5][i]) + (a2 * coeffs[4][i]);
121 coeffs[7][i] += (a1 * coeffs[6][i]) + (a2 * coeffs[5][i]);
122
123 /*
124 * transpose
125 */
126 newCoef->coeffs[i][0] = (float16_t) coeffs[0][i];
127 newCoef->coeffs[i][1] = (float16_t) coeffs[1][i];
128 newCoef->coeffs[i][2] = (float16_t) coeffs[2][i];
129 newCoef->coeffs[i][3] = (float16_t) coeffs[3][i];
130 newCoef->coeffs[i][4] = (float16_t) coeffs[4][i];
131 newCoef->coeffs[i][5] = (float16_t) coeffs[5][i];
132 newCoef->coeffs[i][6] = (float16_t) coeffs[6][i];
133 newCoef->coeffs[i][7] = (float16_t) coeffs[7][i];
134
135 }
136}

138void arm_biquad_cascade_df1_mve_init_f16(arm_biquad_casd_df1_inst_f16 * S,
139 uint8_t numStages,
140 const float16_t * pCoeffs,
141 arm_biquad_mod_coef_f16 * pCoeffsMod,
142 float16_t * pState)
143{
144 arm_biquad_cascade_df1_init_f16(S, numStages, (float16_t *)pCoeffsMod, pState);
145
146 /* Generate SIMD friendly modified coefs */
147 for (int i = 0; i < numStages; i++)
148 {
149 generateCoefsFastBiquadF16(pCoeffs[0], pCoeffs[1], pCoeffs[2], pCoeffs[3], pCoeffs[4], pCoeffsMod);
150 pCoeffs += 5;
151 pCoeffsMod++;
152 }
153}

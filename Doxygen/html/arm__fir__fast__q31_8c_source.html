62void arm_fir_fast_q31(
63 const arm_fir_instance_q31 * S,
64 const q31_t * pSrc,
65 q31_t * pDst,
66 uint32_t blockSize)
67{
68 q31_t *pState = S->pState; /* State pointer */
69 const q31_t *pCoeffs = S->pCoeffs; /* Coefficient pointer */
70 q31_t *pStateCurnt; /* Points to the current sample of the state */
71 q31_t *px; /* Temporary pointer for state buffer */
72 const q31_t *pb; /* Temporary pointer for coefficient buffer */
73 q31_t acc0; /* Accumulators */
74 uint32_t numTaps = S->numTaps; /* Number of filter coefficients in the filter */
75 uint32_t i, tapCnt, blkCnt; /* Loop counters */
76
77#if defined (ARM_MATH_LOOPUNROLL)
78 q31_t acc1, acc2, acc3; /* Accumulators */
79 q31_t x0, x1, x2, x3, c0; /* Temporary variables to hold state and coefficient values */
80#endif
81
82 /* S->pState points to state array which contains previous frame (numTaps - 1) samples */
83 /* pStateCurnt points to the location where the new input data should be written */
84 pStateCurnt = &(S->pState[(numTaps - 1U)]);
85
86#if defined (ARM_MATH_LOOPUNROLL)
87
88 /* Loop unrolling: Compute 4 output values simultaneously.
89 * The variables acc0 ... acc3 hold output values that are being computed:
90 *
91 * acc0 = b[numTaps-1] * x[n-numTaps-1] + b[numTaps-2] * x[n-numTaps-2] + b[numTaps-3] * x[n-numTaps-3] +...+ b[0] * x[0]
92 * acc1 = b[numTaps-1] * x[n-numTaps] + b[numTaps-2] * x[n-numTaps-1] + b[numTaps-3] * x[n-numTaps-2] +...+ b[0] * x[1]
93 * acc2 = b[numTaps-1] * x[n-numTaps+1] + b[numTaps-2] * x[n-numTaps] + b[numTaps-3] * x[n-numTaps-1] +...+ b[0] * x[2]
94 * acc3 = b[numTaps-1] * x[n-numTaps+2] + b[numTaps-2] * x[n-numTaps+1] + b[numTaps-3] * x[n-numTaps] +...+ b[0] * x[3]
95 */
96 blkCnt = blockSize >> 2U;
97
98 while (blkCnt > 0U)
99 {
100 /* Copy 4 new input samples into the state buffer. */
101 *pStateCurnt++ = *pSrc++;
102 *pStateCurnt++ = *pSrc++;
103 *pStateCurnt++ = *pSrc++;
104 *pStateCurnt++ = *pSrc++;
105
106 /* Set all accumulators to zero */
107 acc0 = 0;
108 acc1 = 0;
109 acc2 = 0;
110 acc3 = 0;
111
112 /* Initialize state pointer */
113 px = pState;
114
115 /* Initialize coefficient pointer */
116 pb = pCoeffs;
117
118 /* Read the first 3 samples from the state buffer:
119 * x[n-numTaps], x[n-numTaps-1], x[n-numTaps-2] */
120 x0 = *px++;
121 x1 = *px++;
122 x2 = *px++;
123
124 /* Loop unrolling. Process 4 taps at a time. */
125 tapCnt = numTaps >> 2U;
126
127 /* Loop over the number of taps. Unroll by a factor of 4.
128 Repeat until we've computed numTaps-4 coefficients. */
129 while (tapCnt > 0U)
130 {
131 /* Read the b[numTaps] coefficient */
132 c0 = *pb;
133
134 /* Read x[n-numTaps-3] sample */
135 x3 = *px;
136
137 /* acc0 += b[numTaps] * x[n-numTaps] */
138 multAcc_32x32_keep32_R(acc0, x0, c0);
139
140 /* acc1 += b[numTaps] * x[n-numTaps-1] */
141 multAcc_32x32_keep32_R(acc1, x1, c0);
142
143 /* acc2 += b[numTaps] * x[n-numTaps-2] */
144 multAcc_32x32_keep32_R(acc2, x2, c0);
145
146 /* acc3 += b[numTaps] * x[n-numTaps-3] */
147 multAcc_32x32_keep32_R(acc3, x3, c0);
148
149 /* Read the b[numTaps-1] coefficient */
150 c0 = *(pb + 1U);
151
152 /* Read x[n-numTaps-4] sample */
153 x0 = *(px + 1U);
154
155 /* Perform the multiply-accumulates */
156 multAcc_32x32_keep32_R(acc0, x1, c0);
157 multAcc_32x32_keep32_R(acc1, x2, c0);
158 multAcc_32x32_keep32_R(acc2, x3, c0);
159 multAcc_32x32_keep32_R(acc3, x0, c0);
160
161 /* Read the b[numTaps-2] coefficient */
162 c0 = *(pb + 2U);
163
164 /* Read x[n-numTaps-5] sample */
165 x1 = *(px + 2U);
166
167 /* Perform the multiply-accumulates */
168 multAcc_32x32_keep32_R(acc0, x2, c0);
169 multAcc_32x32_keep32_R(acc1, x3, c0);
170 multAcc_32x32_keep32_R(acc2, x0, c0);
171 multAcc_32x32_keep32_R(acc3, x1, c0);
172
173 /* Read the b[numTaps-3] coefficients */
174 c0 = *(pb + 3U);
175
176 /* Read x[n-numTaps-6] sample */
177 x2 = *(px + 3U);
178
179 /* Perform the multiply-accumulates */
180 multAcc_32x32_keep32_R(acc0, x3, c0);
181 multAcc_32x32_keep32_R(acc1, x0, c0);
182 multAcc_32x32_keep32_R(acc2, x1, c0);
183 multAcc_32x32_keep32_R(acc3, x2, c0);
184
185 /* update coefficient pointer */
186 pb += 4U;
187 px += 4U;
188
189 /* Decrement loop counter */
190 tapCnt--;
191 }
192
193 /* If the filter length is not a multiple of 4, compute the remaining filter taps */
194 tapCnt = numTaps % 0x4U;
195
196 while (tapCnt > 0U)
197 {
198 /* Read coefficients */
199 c0 = *(pb++);
200
201 /* Fetch 1 state variable */
202 x3 = *(px++);
203
204 /* Perform the multiply-accumulates */
205 multAcc_32x32_keep32_R(acc0, x0, c0);
206 multAcc_32x32_keep32_R(acc1, x1, c0);
207 multAcc_32x32_keep32_R(acc2, x2, c0);
208 multAcc_32x32_keep32_R(acc3, x3, c0);
209
210 /* Reuse the present sample states for next sample */
211 x0 = x1;
212 x1 = x2;
213 x2 = x3;
214
215 /* Decrement loop counter */
216 tapCnt--;
217 }
218
219 /* The results in the 4 accumulators are in 2.30 format. Convert to 1.31
220 Then store the 4 outputs in the destination buffer. */
221 *pDst++ = (q31_t) (acc0 << 1);
222 *pDst++ = (q31_t) (acc1 << 1);
223 *pDst++ = (q31_t) (acc2 << 1);
224 *pDst++ = (q31_t) (acc3 << 1);
225
226 /* Advance the state pointer by 4 to process the next group of 4 samples */
227 pState = pState + 4U;
228
229 /* Decrement loop counter */
230 blkCnt--;
231 }
232
233 /* Loop unrolling: Compute remaining output samples */
234 blkCnt = blockSize % 0x4U;
235
236#else
237
238 /* Initialize blkCnt with number of taps */
239 blkCnt = blockSize;
240
241#endif /* #if defined (ARM_MATH_LOOPUNROLL) */
242
243 while (blkCnt > 0U)
244 {
245 /* Copy one sample at a time into state buffer */
246 *pStateCurnt++ = *pSrc++;
247
248 /* Set the accumulator to zero */
249 acc0 = 0;
250
251 /* Initialize state pointer */
252 px = pState;
253
254 /* Initialize Coefficient pointer */
255 pb = pCoeffs;
256
257 i = numTaps;
258
259 /* Perform the multiply-accumulates */
260 do
261 {
262 multAcc_32x32_keep32_R(acc0, (*px++), (*pb++));
263 i--;
264 } while (i > 0U);
265
266 /* The result is in 2.30 format. Convert to 1.31
267 Then store the output in the destination buffer. */
268 *pDst++ = (q31_t) (acc0 << 1);
269
270 /* Advance state pointer by 1 for the next sample */
271 pState = pState + 1U;
272
273 /* Decrement loop counter */
274 blkCnt--;
275 }
276
277 /* Processing is complete.
278 Now copy the last numTaps - 1 samples to the start of the state buffer.
279 This prepares the state buffer for the next function call. */
280
281 /* Points to the start of the state buffer */
282 pStateCurnt = S->pState;
283
284#if defined (ARM_MATH_LOOPUNROLL)
285
286 /* Loop unrolling: Compute 4 taps at a time */
287 tapCnt = (numTaps - 1U) >> 2U;
288
289 /* Copy data */
290 while (tapCnt > 0U)
291 {
292 *pStateCurnt++ = *pState++;
293 *pStateCurnt++ = *pState++;
294 *pStateCurnt++ = *pState++;
295 *pStateCurnt++ = *pState++;
296
297 /* Decrement loop counter */
298 tapCnt--;
299 }
300
301 /* Calculate remaining number of copies */
302 tapCnt = (numTaps - 1U) % 0x4U;
303
304#else
305
306 /* Initialize tapCnt with number of taps */
307 tapCnt = (numTaps - 1U);
308
309#endif /* #if defined (ARM_MATH_LOOPUNROLL) */
310
311 /* Copy remaining data */
312 while (tapCnt > 0U)
313 {
314 *pStateCurnt++ = *pState++;
315
316 /* Decrement the loop counter */
317 tapCnt--;
318 }
319
320}

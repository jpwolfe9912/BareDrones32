202 accelSum[YAXIS] += ((float)rawAccel[YAXIS].value - eepromConfig.accelBiasMPU[YAXIS] - accelTCBias[YAXIS]) * eepromConfig.accelScaleFactorMPU[YAXIS];
203 accelSum[ZAXIS] += ((float)rawAccel[ZAXIS].value - eepromConfig.accelBiasMPU[ZAXIS] - accelTCBias[ZAXIS]) * eepromConfig.accelScaleFactorMPU[ZAXIS];
204
205 gyroSum[ROLL ] += (float)rawGyro[ROLL ].value - gyroTCBias[ROLL ];
206 gyroSum[PITCH] += (float)rawGyro[PITCH].value - gyroTCBias[PITCH];
207 gyroSum[YAW ] += (float)rawGyro[YAW ].value - gyroTCBias[YAW ];
208
209 delayMicroseconds(100);
210 }
211
212 for (axis = 0; axis < 3; axis++)
213 {
214 accelSum[axis] = accelSum[axis] / 5000.0f;
215 gyroRTBias[axis] = gyroSum[axis] / 5000.0f;
216 }
217
218 accelOneG = sqrt(SQR(accelSum[XAXIS]) + SQR(accelSum[YAXIS]) + SQR(accelSum[ZAXIS]));

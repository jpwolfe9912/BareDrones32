77void arm_fir_init_f16(
78 arm_fir_instance_f16 * S,
79 uint16_t numTaps,
80 const float16_t * pCoeffs,
81 float16_t * pState,
82 uint32_t blockSize)
83{

86
87 /* Assign coefficient pointer */
88 S->pCoeffs = pCoeffs;
89
90 /* Clear state buffer. The size is always (blockSize + numTaps - 1) */
91#if defined(ARM_MATH_MVE_FLOAT16) && !defined(ARM_MATH_AUTOVECTORIZE)
92 memset(pState, 0, (numTaps + (blockSize - 1U) + ROUND_UP(blockSize, 8)) * sizeof(float16_t));
93#else
94 memset(pState, 0, (numTaps + (blockSize - 1U)) * sizeof(float16_t));
95#endif
96
97 /* Assign state pointer */
98 S->pState = pState;
99}

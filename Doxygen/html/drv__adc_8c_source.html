123static battCells_e
124batt_cells(void)
125{
126 uint16_t summedVoltage = 0;
127 uint16_t avgVoltage = 0;
128 for(uint8_t i = 0; i < 10; i++)
129 {
130 batMonRead();
131 summedVoltage += *rawBatt;
132 delay(1);
133 }
134 avgVoltage = summedVoltage / 10;
135
136 if((avgVoltage >= 1128) && (avgVoltage < 1501))
137 return THREE;
138 else if((avgVoltage >= 1501) && (avgVoltage < 1935))
139 return FOUR;
140 else if((avgVoltage >= 1935) && (avgVoltage < 2607))
141 return FIVE;
142 else if(avgVoltage >= 2607)
143 return SIX;
144 else
145 return NO_BATT;
146
147}

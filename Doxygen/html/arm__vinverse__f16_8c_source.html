37void arm_vinverse_f16(
38 const float16_t * pSrc,
39 float16_t * pDst,
40 uint32_t blockSize)
41{
42 uint32_t blkCnt;
43
44#if defined(ARM_MATH_MVE_FLOAT16) && !defined(ARM_MATH_AUTOVECTORIZE)
45
46 f16x8_t src;
47 f16x8_t dst;
48
49 blkCnt = blockSize >> 3;
50
51 while (blkCnt > 0U)
52 {
53 src = vld1q(pSrc);
54 dst = vrecip_hiprec_f16(src);
55 vst1q(pDst, dst);
56
57 pSrc += 8;
58 pDst += 8;
59 /* Decrement loop counter */
60 blkCnt--;
61 }
62
63 blkCnt = blockSize & 7;
64#else
65 blkCnt = blockSize;
66#endif
67
68 while (blkCnt > 0U)
69 {
70
71 *pDst++ = 1.0 / *pSrc++;
72
73 /* Decrement loop counter */
74 blkCnt--;
75 }
76}
77
78#endif /* #if defined(ARM_FLOAT16_SUPPORTED) */

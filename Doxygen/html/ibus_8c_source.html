100 uint8_t b = rxBuf[framePos++];
101
102 status = IBUS_BUSY;
103 switch(state){
104 case IBUS_STATE_LENGTH: { // length byte
105 if(b == 0x20){
106
107 frameLength = b;
108 state++;
109 }
110 else
111 {
112 status = IBUS_ERROR;
113 }
114 break;
115 }
116 case IBUS_STATE_TYPE: { // type byte
117 if(b == 0x40){
118 devID = b;
119 state++;
120 }
121 else
122 status = IBUS_ERROR;
123 break;
124 }
125 case IBUS_STATE_PAYLOAD: { // payload bytes
126 ibusPayload[payloadPos++] = b;
127 if(payloadPos == PAYLOAD_SIZE)
128 {
129 ibus_update(ibusPayload);
130 state++;
131 payloadPos = 0;
132 }
133 break;
134 }
135 case IBUS_STATE_CRC: { // crc bytes
136 ibusCRC[crcPos++] = b;
137 if(crcPos == CRC_SIZE){
138 if(ibus_frame_crc(ibusCRC))
139 {
140 status = IBUS_READY;
141 }
142 else{
143 status = IBUS_ERROR;
144 }
145 crcPos = 0;
146 }
147 break;
148 }
149 }
150 if(status == IBUS_ERROR)
151 {
152 framePos = 0;
153 state = 0;
154 }
155 if(status == IBUS_READY)
156 {
157 framePos = 0;
158 state = 0;
159 }
160
161 return status;

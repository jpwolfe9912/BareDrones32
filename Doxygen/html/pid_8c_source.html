40float
41updatePID(float error, float deltaT, uint8_t reset, struct PIDdata *PIDparameters)
42{
43 float dTerm;
44 float pidSum;
45 float pidLimited;
46
47 if ((reset == true) || (PIDparameters->prevResetState == true))
48 {
49 PIDparameters->integratorState = 0.0f;
50 PIDparameters->filterState = 0.0f;
51 }
52
53 dTerm = ((error * PIDparameters->D) - PIDparameters->filterState) * 100.0f;
54
55 pidSum = (error * PIDparameters->P) + PIDparameters->integratorState + dTerm;
56
57 if (pidSum > PIDparameters->Limit)
58 {
59 pidLimited = PIDparameters->Limit;
60 }
61 else
62 {
63 pidLimited = -PIDparameters->Limit;
64
65 if (!(pidSum < (-PIDparameters->Limit)))
66 {
67 pidLimited = pidSum;
68 }
69 }
70
71 PIDparameters->integratorState += ((error * PIDparameters->I) + 100.0f * (pidLimited - pidSum)) * deltaT;
72
73 PIDparameters->filterState += deltaT * dTerm;
74
75 if (reset == true)
76 PIDparameters->prevResetState = true;
77 else
78 PIDparameters->prevResetState = false;
79
80 return pidLimited;
81}

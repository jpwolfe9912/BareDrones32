66void arm_lms_q31(
67 const arm_lms_instance_q31 * S,
68 const q31_t * pSrc,
69 q31_t * pRef,
70 q31_t * pOut,
71 q31_t * pErr,
72 uint32_t blockSize)
73{
74 q31_t *pState = S->pState; /* State pointer */
75 q31_t *pCoeffs = S->pCoeffs; /* Coefficient pointer */
76 q31_t *pStateCurnt; /* Points to the current sample of the state */
77 q31_t *px, *pb; /* Temporary pointers for state and coefficient buffers */
78 q31_t mu = S->mu; /* Adaptive factor */
79 uint32_t numTaps = S->numTaps; /* Number of filter coefficients in the filter */
80 uint32_t tapCnt, blkCnt; /* Loop counters */
81 q63_t acc; /* Accumulator */
82 q31_t e = 0; /* Error of data sample */
83 q31_t alpha; /* Intermediate constant for taps update */
84 q31_t coef; /* Temporary variable for coef */
85 q31_t acc_l, acc_h; /* Temporary input */
86 uint32_t uShift = ((uint32_t) S->postShift + 1U);
87 uint32_t lShift = 32U - uShift; /* Shift to be applied to the output */
88
89 /* S->pState points to buffer which contains previous frame (numTaps - 1) samples */
90 /* pStateCurnt points to the location where the new input data should be written */
91 pStateCurnt = &(S->pState[(numTaps - 1U)]);
92
93 /* initialise loop count */
94 blkCnt = blockSize;
95
96 while (blkCnt > 0U)
97 {
98 /* Copy the new input sample into the state buffer */
99 *pStateCurnt++ = *pSrc++;
100
101 /* Initialize pState pointer */
102 px = pState;
103
104 /* Initialize coefficient pointer */
105 pb = pCoeffs;
106
107 /* Set the accumulator to zero */
108 acc = 0;
109
110#if defined (ARM_MATH_LOOPUNROLL)
111
112 /* Loop unrolling: Compute 4 taps at a time. */
113 tapCnt = numTaps >> 2U;
114
115 while (tapCnt > 0U)
116 {
117 /* Perform the multiply-accumulate */
118 /* acc += b[N] * x[n-N] */
119 acc += ((q63_t) (*px++)) * (*pb++);
120
121 /* acc += b[N-1] * x[n-N-1] */
122 acc += ((q63_t) (*px++)) * (*pb++);
123
124 /* acc += b[N-2] * x[n-N-2] */
125 acc += ((q63_t) (*px++)) * (*pb++);
126
127 /* acc += b[N-3] * x[n-N-3] */
128 acc += ((q63_t) (*px++)) * (*pb++);
129
130 /* Decrement loop counter */
131 tapCnt--;
132 }
133
134 /* Loop unrolling: Compute remaining taps */
135 tapCnt = numTaps % 0x4U;
136
137#else
138
139 /* Initialize tapCnt with number of samples */
140 tapCnt = numTaps;
141
142#endif /* #if defined (ARM_MATH_LOOPUNROLL) */
143
144 while (tapCnt > 0U)
145 {
146 /* Perform the multiply-accumulate */
147 acc += ((q63_t) (*px++)) * (*pb++);
148
149 /* Decrement the loop counter */
150 tapCnt--;
151 }
152
153 /* Converting the result to 1.31 format */
154 /* Calc lower part of acc */
155 acc_l = acc & 0xffffffff;
156
157 /* Calc upper part of acc */
158 acc_h = (acc >> 32) & 0xffffffff;
159
160 acc = (uint32_t) acc_l >> lShift | acc_h << uShift;
161
162 /* Store the result from accumulator into the destination buffer. */
163 *pOut++ = (q31_t) acc;
164
165 /* Compute and store error */
166 e = *pRef++ - (q31_t) acc;
167 *pErr++ = e;
168
169 /* Compute alpha i.e. intermediate constant for taps update */
170 alpha = (q31_t) (((q63_t) e * mu) >> 31);
171
172 /* Initialize pState pointer */
173 /* Advance state pointer by 1 for the next sample */
174 px = pState++;
175
176 /* Initialize coefficient pointer */
177 pb = pCoeffs;
178
179#if defined (ARM_MATH_LOOPUNROLL)
180
181 /* Loop unrolling: Compute 4 taps at a time. */
182 tapCnt = numTaps >> 2U;
183
184 /* Update filter coefficients */
185 while (tapCnt > 0U)
186 {
187 /* Perform the multiply-accumulate */
188
189 /* coef is in 2.30 format */
190 coef = (q31_t) (((q63_t) alpha * (*px++)) >> (32));
191 /* get coef in 1.31 format by left shifting */
192 *pb = clip_q63_to_q31((q63_t) * pb + (coef << 1U));
193 /* update coefficient buffer to next coefficient */
194 pb++;
195
196 coef = (q31_t) (((q63_t) alpha * (*px++)) >> (32));
197 *pb = clip_q63_to_q31((q63_t) * pb + (coef << 1U));
198 pb++;
199
200 coef = (q31_t) (((q63_t) alpha * (*px++)) >> (32));
201 *pb = clip_q63_to_q31((q63_t) * pb + (coef << 1U));
202 pb++;
203
204 coef = (q31_t) (((q63_t) alpha * (*px++)) >> (32));
205 *pb = clip_q63_to_q31((q63_t) * pb + (coef << 1U));
206 pb++;
207
208 /* Decrement loop counter */
209 tapCnt--;
210 }
211
212 /* Loop unrolling: Compute remaining taps */
213 tapCnt = numTaps % 0x4U;
214
215#else
216
217 /* Initialize tapCnt with number of samples */
218 tapCnt = numTaps;
219
220#endif /* #if defined (ARM_MATH_LOOPUNROLL) */
221
222 while (tapCnt > 0U)
223 {
224 /* Perform the multiply-accumulate */
225 coef = (q31_t) (((q63_t) alpha * (*px++)) >> (32));
226 *pb = clip_q63_to_q31((q63_t) * pb + (coef << 1U));
227 pb++;
228
229 /* Decrement loop counter */
230 tapCnt--;
231 }
232
233 /* Decrement loop counter */
234 blkCnt--;
235 }
236
237 /* Processing is complete.
238 Now copy the last numTaps - 1 samples to the start of the state buffer.
239 This prepares the state buffer for the next function call. */
240
241 /* Points to the start of the pState buffer */
242 pStateCurnt = S->pState;
243
244 /* copy data */
245#if defined (ARM_MATH_LOOPUNROLL)
246
247 /* Loop unrolling: Compute 4 taps at a time. */
248 tapCnt = (numTaps - 1U) >> 2U;
249
250 while (tapCnt > 0U)
251 {
252 *pStateCurnt++ = *pState++;
253 *pStateCurnt++ = *pState++;
254 *pStateCurnt++ = *pState++;
255 *pStateCurnt++ = *pState++;
256
257 /* Decrement loop counter */
258 tapCnt--;
259 }
260
261 /* Loop unrolling: Compute remaining taps */
262 tapCnt = (numTaps - 1U) % 0x4U;
263
264#else
265
266 /* Initialize tapCnt with number of samples */
267 tapCnt = (numTaps - 1U);
268
269#endif /* #if defined (ARM_MATH_LOOPUNROLL) */
270
271 while (tapCnt > 0U)
272 {
273 *pStateCurnt++ = *pState++;
274
275 /* Decrement loop counter */
276 tapCnt--;
277 }
278
279}

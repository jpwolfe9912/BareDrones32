167void arm_biquad_cascade_df1_f32(
168 const arm_biquad_casd_df1_inst_f32 * S,
169 const float32_t * pSrc,
170 float32_t * pDst,
171 uint32_t blockSize)
172{
173 const float32_t *pIn = pSrc; /* source pointer */
174 float32_t *pOut = pDst; /* destination pointer */
175 float32_t *pState = S->pState; /* pState pointer */
176 const float32_t *pCoeffs = S->pCoeffs; /* coefficient pointer */
177 float32_t Xn1, Xn2, Yn1, Yn2; /* Filter pState variables */
178 float32_t lastX, lastY; /* X,Y history for tail handling */
179 float32_t X0, X1, X2, X3; /* temporary input */
180 f32x4_t coeffs;
181 f32x4_t accVec; /* accumultor vector */
182 uint32_t sample, stage = S->numStages; /* loop counters */
183
184 do
185 {
186 /*
187 * Reading the pState values
188 */
189 Xn1 = pState[0];
190 Xn2 = pState[1];
191 Yn1 = pState[2];
192 Yn2 = pState[3];
193
194 sample = blockSize >> 2U;
195
196 /*
197 * First part of the processing with loop unrolling. Compute 4 outputs at a time.
198 * second loop below computes the remaining 1 to 3 samples.
199 */
200 while (sample > 0U)
201 {
202 X0 = *pIn++;
203 X1 = *pIn++;
204 X2 = *pIn++;
205 X3 = *pIn++;
206
207 coeffs = vld1q(pCoeffs);
208 accVec = vmulq(coeffs, X3);
209
210 coeffs = vld1q(&pCoeffs[4]);
211 accVec = vfmaq(accVec, coeffs, X2);
212
213 coeffs = vld1q(&pCoeffs[8]);
214 accVec = vfmaq(accVec, coeffs, X1);
215
216 coeffs = vld1q(&pCoeffs[12]);
217 accVec = vfmaq(accVec, coeffs, X0);
218
219 coeffs = vld1q(&pCoeffs[16]);
220 accVec = vfmaq(accVec, coeffs, Xn1);
221
222 coeffs = vld1q(&pCoeffs[20]);
223 accVec = vfmaq(accVec, coeffs, Xn2);
224
225 coeffs = vld1q(&pCoeffs[24]);
226 accVec = vfmaq(accVec, coeffs, Yn1);
227
228 coeffs = vld1q(&pCoeffs[28]);
229 accVec = vfmaq(accVec, coeffs, Yn2);
230 /*
231 * Store the result in the accumulator in the destination buffer.
232 */
233 vst1q(pOut, accVec);
234 pOut += 4;
235
236 /*
237 * update recurrence
238 */
239 Xn1 = X3;
240 Xn2 = X2;
241 Yn1 = vgetq_lane(accVec, 3);
242 Yn2 = vgetq_lane(accVec, 2);
243 /*
244 * decrement the loop counter
245 */
246 sample--;
247 }
248
249 /*
250 * If the blockSize is not a multiple of 4,
251 * compute any remaining output samples here.
252 */
253 sample = blockSize & 0x3U;
254 if (sample)
255 {
256 /* save previous X, Y for modulo 1 length case */
257 lastX = X3;
258 lastY = Yn1;
259
260 X0 = *pIn++;
261 X1 = *pIn++;
262 X2 = *pIn++;
263 X3 = *pIn++;
264
265 coeffs = vld1q(pCoeffs);
266 accVec = vmulq(coeffs, X3);
267
268 coeffs = vld1q(&pCoeffs[4]);
269 accVec = vfmaq(accVec, coeffs, X2);
270
271 coeffs = vld1q(&pCoeffs[8]);
272 accVec = vfmaq(accVec, coeffs, X1);
273
274 coeffs = vld1q(&pCoeffs[12]);
275 accVec = vfmaq(accVec, coeffs, X0);
276
277 coeffs = vld1q(&pCoeffs[16]);
278 accVec = vfmaq(accVec, coeffs, Xn1);
279
280 coeffs = vld1q(&pCoeffs[20]);
281 accVec = vfmaq(accVec, coeffs, Xn2);
282
283 coeffs = vld1q(&pCoeffs[24]);
284 accVec = vfmaq(accVec, coeffs, Yn1);
285
286 coeffs = vld1q(&pCoeffs[28]);
287 accVec = vfmaq(accVec, coeffs, Yn2);
288
289 if (sample == 1)
290 {
291 *pOut++ = vgetq_lane(accVec, 0);
292 Xn1 = X0;
293 Xn2 = lastX;
294 Yn1 = vgetq_lane(accVec, 0);
295 Yn2 = lastY;
296 }
297 else if (sample == 2)
298 {
299 *pOut++ = vgetq_lane(accVec, 0);
300 *pOut++ = vgetq_lane(accVec, 1);
301 Xn1 = X1;
302 Xn2 = X0;
303 Yn1 = vgetq_lane(accVec, 1);
304 Yn2 = vgetq_lane(accVec, 0);
305 }
306 else
307 {
308 *pOut++ = vgetq_lane(accVec, 0);
309 *pOut++ = vgetq_lane(accVec, 1);
310 *pOut++ = vgetq_lane(accVec, 2);
311 Xn1 = X2;
312 Xn2 = X1;
313 Yn1 = vgetq_lane(accVec, 2);
314 Yn2 = vgetq_lane(accVec, 1);
315 }
316 }
317 /*
318 * Store the updated state variables back into the pState array
319 */
320 *pState++ = Xn1;
321 *pState++ = Xn2;
322 *pState++ = Yn1;
323 *pState++ = Yn2;
324
325 pCoeffs += sizeof(arm_biquad_mod_coef_f32) / sizeof(float32_t);
326 /*
327 * The first stage goes from the input buffer to the output buffer.
328 * Subsequent numStages occur in-place in the output buffer
329 */
330 pIn = pDst;
331 /*
332 * Reset the output pointer
333 */
334 pOut = pDst;
335 /*
336 * decrement the loop counter
337 */
338 stage--;
339 }
340 while (stage > 0U);
341}

344void arm_biquad_cascade_df1_f32(
345 const arm_biquad_casd_df1_inst_f32 * S,
346 const float32_t * pSrc,
347 float32_t * pDst,
348 uint32_t blockSize)
349{
350
351 const float32_t *pIn = pSrc; /* source pointer */
352 float32_t *pOut = pDst; /* destination pointer */
353 float32_t *pState = S->pState; /* pState pointer */
354 const float32_t *pCoeffs = S->pCoeffs; /* coefficient pointer */
355 float32_t acc; /* Simulates the accumulator */
356
357 uint32_t sample, stage = S->numStages; /* loop counters */
358
359 float32x4_t Xn;
360 float32x2_t Yn;
361 float32x2_t a;
362 float32x4_t b;
363
364 float32x4_t x,tmp;
365 float32x2_t t;
366 float32x2x2_t y;
367
368 float32_t Xns;
369
370 while (stage > 0U)
371 {
372 /* Reading the coefficients */
373 Xn = vdupq_n_f32(0.0f);
374
375 Xn = vsetq_lane_f32(pState[0],Xn,2);
376 Xn = vsetq_lane_f32(pState[1],Xn,3);
377 Yn = vset_lane_f32(pState[2],Yn,0);
378 Yn = vset_lane_f32(pState[3],Yn,1);
379
380 b = vld1q_f32(pCoeffs);
381 b = vrev64q_f32(b);
382 b = vcombine_f32(vget_high_f32(b), vget_low_f32(b));
383
384 a = vld1_f32(pCoeffs + 3);
385 a = vrev64_f32(a);
386 b = vsetq_lane_f32(0.0f,b,0);
387 pCoeffs += 5;
388
389 /* Reading the pState values */
390
391 /* Apply loop unrolling and compute 4 output values simultaneously. */
392 /* The variable acc hold output values that are being computed:
393 *
394 * acc = b0 * x[n] + b1 * x[n-1] + b2 * x[n-2] + a1 * y[n-1] + a2 * y[n-2]
395 * acc = b0 * x[n] + b1 * x[n-1] + b2 * x[n-2] + a1 * y[n-1] + a2 * y[n-2]
396 * acc = b0 * x[n] + b1 * x[n-1] + b2 * x[n-2] + a1 * y[n-1] + a2 * y[n-2]
397 * acc = b0 * x[n] + b1 * x[n-1] + b2 * x[n-2] + a1 * y[n-1] + a2 * y[n-2]
398 */
399
400 /* First part of the processing with loop unrolling. Compute 4 outputs at a time.
401 ** a second loop below computes the remaining 1 to 3 samples. */
402 sample = blockSize >> 2U;
403
404 while (sample > 0U)
405 {
406 /* Read the first 4 inputs */
407 x = vld1q_f32(pIn);
408
409 pIn += 4;
410
411 tmp = vextq_f32(Xn, x, 1);
412 t = vmul_f32(vget_high_f32(b), vget_high_f32(tmp));
413 t = vmla_f32(t, vget_low_f32(b), vget_low_f32(tmp));
414 t = vmla_f32(t, a, Yn);
415 t = vpadd_f32(t, t);
416 Yn = vext_f32(Yn, t, 1);
417
418 tmp = vextq_f32(Xn, x, 2);
419 t = vmul_f32(vget_high_f32(b), vget_high_f32(tmp));
420 t = vmla_f32(t, vget_low_f32(b), vget_low_f32(tmp));
421 t = vmla_f32(t, a, Yn);
422 t = vpadd_f32(t, t);
423 Yn = vext_f32(Yn, t, 1);
424
425 y.val[0] = Yn;
426
427 tmp = vextq_f32(Xn, x, 3);
428 t = vmul_f32(vget_high_f32(b), vget_high_f32(tmp));
429 t = vmla_f32(t, vget_low_f32(b), vget_low_f32(tmp));
430 t = vmla_f32(t, a, Yn);
431 t = vpadd_f32(t, t);
432 Yn = vext_f32(Yn, t, 1);
433
434 Xn = x;
435 t = vmul_f32(vget_high_f32(b), vget_high_f32(Xn));
436 t = vmla_f32(t, vget_low_f32(b), vget_low_f32(Xn));
437 t = vmla_f32(t, a, Yn);
438 t = vpadd_f32(t, t);
439 Yn = vext_f32(Yn, t, 1);
440
441 y.val[1] = Yn;
442
443 tmp = vcombine_f32(y.val[0], y.val[1]);
444
445 /* Store the 4 outputs and increment the pointer */
446 vst1q_f32(pOut, tmp);
447 pOut += 4;
448
449 /* Decrement the loop counter */
450 sample--;
451 }
452
453
454 /* If the block size is not a multiple of 4, compute any remaining output samples here.
455 ** No loop unrolling is used. */
456 sample = blockSize & 0x3U;
457
458 while (sample > 0U)
459 {
460 /* Read the input */
461 Xns = *pIn++;
462
463 /* acc = b0 * x[n] + b1 * x[n-1] + b2 * x[n-2] + a1 * y[n-1] + a2 * y[n-2] */
464 acc = (vgetq_lane_f32(b, 1) * vgetq_lane_f32(Xn, 2))
465 + (vgetq_lane_f32(b, 2) * vgetq_lane_f32(Xn, 3))
466 + (vgetq_lane_f32(b, 3) * Xns)
467 + (vget_lane_f32(a, 0) * vget_lane_f32(Yn, 0))
468 + (vget_lane_f32(a, 1) * vget_lane_f32(Yn, 1));
469
470 /* Store the result in the accumulator in the destination buffer. */
471 *pOut++ = acc;
472
473 /* Every time after the output is computed state should be updated. */
474 /* The states should be updated as: */
475 /* Xn2 = Xn1 */
476 /* Xn1 = Xn */
477 /* Yn2 = Yn1 */
478 /* Yn1 = acc */
479 Xn = vsetq_lane_f32(vgetq_lane_f32(Xn, 3),Xn,2);
480 Xn = vsetq_lane_f32(Xns,Xn,3);
481 Yn = vset_lane_f32(vget_lane_f32(Yn, 1),Yn,0);
482 Yn = vset_lane_f32(acc,Yn,1);
483
484 /* Decrement the loop counter */
485 sample--;
486
487 }
488
489 vst1q_f32(pState,vcombine_f32((vget_high_f32(Xn)),(Yn)));
490 pState += 4;
491 /* Store the updated state variables back into the pState array */
492
493 /* The first stage goes from the input buffer to the output buffer. */
494 /* Subsequent numStages occur in-place in the output buffer */
495 pIn = pDst;
496
497 /* Reset the output pointer */
498 pOut = pDst;
499
500 /* Decrement the loop counter */
501 stage--;
502 }
503}

506void arm_biquad_cascade_df1_f32(
507 const arm_biquad_casd_df1_inst_f32 * S,
508 const float32_t * pSrc,
509 float32_t * pDst,
510 uint32_t blockSize)
511{
512 const float32_t *pIn = pSrc; /* Source pointer */
513 float32_t *pOut = pDst; /* Destination pointer */
514 float32_t *pState = S->pState; /* pState pointer */
515 const float32_t *pCoeffs = S->pCoeffs; /* Coefficient pointer */
516 float32_t acc; /* Accumulator */
517 float32_t b0, b1, b2, a1, a2; /* Filter coefficients */
518 float32_t Xn1, Xn2, Yn1, Yn2; /* Filter pState variables */
519 float32_t Xn; /* Temporary input */
520 uint32_t sample, stage = S->numStages; /* Loop counters */
521
522 do
523 {
524 /* Reading the coefficients */
525 b0 = *pCoeffs++;
526 b1 = *pCoeffs++;
527 b2 = *pCoeffs++;
528 a1 = *pCoeffs++;
529 a2 = *pCoeffs++;
530
531 /* Reading the pState values */
532 Xn1 = pState[0];
533 Xn2 = pState[1];
534 Yn1 = pState[2];
535 Yn2 = pState[3];
536
537#if defined (ARM_MATH_LOOPUNROLL) && !defined(ARM_MATH_AUTOVECTORIZE)
538
539 /* Apply loop unrolling and compute 4 output values simultaneously. */
540 /* Variable acc hold output values that are being computed:
541 *
542 * acc = b0 * x[n] + b1 * x[n-1] + b2 * x[n-2] + a1 * y[n-1] + a2 * y[n-2]
543 * acc = b0 * x[n] + b1 * x[n-1] + b2 * x[n-2] + a1 * y[n-1] + a2 * y[n-2]
544 * acc = b0 * x[n] + b1 * x[n-1] + b2 * x[n-2] + a1 * y[n-1] + a2 * y[n-2]
545 * acc = b0 * x[n] + b1 * x[n-1] + b2 * x[n-2] + a1 * y[n-1] + a2 * y[n-2]
546 */
547
548 /* Loop unrolling: Compute 4 outputs at a time */
549 sample = blockSize >> 2U;
550
551 while (sample > 0U)
552 {
553 /* Read the first input */
554 Xn = *pIn++;
555
556 /* acc = b0 * x[n] + b1 * x[n-1] + b2 * x[n-2] + a1 * y[n-1] + a2 * y[n-2] */
557 Yn2 = (b0 * Xn) + (b1 * Xn1) + (b2 * Xn2) + (a1 * Yn1) + (a2 * Yn2);
558
559 /* Store output in destination buffer. */
560 *pOut++ = Yn2;
561
562 /* Every time after the output is computed state should be updated. */
563 /* The states should be updated as: */
564 /* Xn2 = Xn1 */
565 /* Xn1 = Xn */
566 /* Yn2 = Yn1 */
567 /* Yn1 = acc */
568
569 /* Read the second input */
570 Xn2 = *pIn++;
571
572 /* acc = b0 * x[n] + b1 * x[n-1] + b2 * x[n-2] + a1 * y[n-1] + a2 * y[n-2] */
573 Yn1 = (b0 * Xn2) + (b1 * Xn) + (b2 * Xn1) + (a1 * Yn2) + (a2 * Yn1);
574
575 /* Store output in destination buffer. */
576 *pOut++ = Yn1;
577
578 /* Every time after the output is computed state should be updated. */
579 /* The states should be updated as: */
580 /* Xn2 = Xn1 */
581 /* Xn1 = Xn */
582 /* Yn2 = Yn1 */
583 /* Yn1 = acc */
584
585 /* Read the third input */
586 Xn1 = *pIn++;
587
588 /* acc = b0 * x[n] + b1 * x[n-1] + b2 * x[n-2] + a1 * y[n-1] + a2 * y[n-2] */
589 Yn2 = (b0 * Xn1) + (b1 * Xn2) + (b2 * Xn) + (a1 * Yn1) + (a2 * Yn2);
590
591 /* Store output in destination buffer. */
592 *pOut++ = Yn2;
593
594 /* Every time after the output is computed state should be updated. */
595 /* The states should be updated as: */
596 /* Xn2 = Xn1 */
597 /* Xn1 = Xn */
598 /* Yn2 = Yn1 */
599 /* Yn1 = acc */
600
601 /* Read the forth input */
602 Xn = *pIn++;
603
604 /* acc = b0 * x[n] + b1 * x[n-1] + b2 * x[n-2] + a1 * y[n-1] + a2 * y[n-2] */
605 Yn1 = (b0 * Xn) + (b1 * Xn1) + (b2 * Xn2) + (a1 * Yn2) + (a2 * Yn1);
606
607 /* Store output in destination buffer. */
608 *pOut++ = Yn1;
609
610 /* Every time after the output is computed state should be updated. */
611 /* The states should be updated as: */
612 /* Xn2 = Xn1 */
613 /* Xn1 = Xn */
614 /* Yn2 = Yn1 */
615 /* Yn1 = acc */
616 Xn2 = Xn1;
617 Xn1 = Xn;
618
619 /* decrement loop counter */
620 sample--;
621 }
622
623 /* Loop unrolling: Compute remaining outputs */
624 sample = blockSize & 0x3U;
625
626#else
627
628 /* Initialize blkCnt with number of samples */
629 sample = blockSize;
630
631#endif /* #if defined (ARM_MATH_LOOPUNROLL) */
632
633 while (sample > 0U)
634 {
635 /* Read the input */
636 Xn = *pIn++;
637
638 /* acc = b0 * x[n] + b1 * x[n-1] + b2 * x[n-2] + a1 * y[n-1] + a2 * y[n-2] */
639 acc = (b0 * Xn) + (b1 * Xn1) + (b2 * Xn2) + (a1 * Yn1) + (a2 * Yn2);
640
641 /* Store output in destination buffer. */
642 *pOut++ = acc;
643
644 /* Every time after the output is computed state should be updated. */
645 /* The states should be updated as: */
646 /* Xn2 = Xn1 */
647 /* Xn1 = Xn */
648 /* Yn2 = Yn1 */
649 /* Yn1 = acc */
650 Xn2 = Xn1;
651 Xn1 = Xn;
652 Yn2 = Yn1;
653 Yn1 = acc;
654
655 /* decrement loop counter */
656 sample--;
657 }
658
659 /* Store the updated state variables back into the pState array */
660 *pState++ = Xn1;
661 *pState++ = Xn2;
662 *pState++ = Yn1;
663 *pState++ = Yn2;
664
665 /* The first stage goes from the input buffer to the output buffer. */
666 /* Subsequent numStages occur in-place in the output buffer */
667 pIn = pDst;
668
669 /* Reset output pointer */
670 pOut = pDst;
671
672 /* decrement loop counter */
673 stage--;
674
675 } while (stage > 0U);
676
677}

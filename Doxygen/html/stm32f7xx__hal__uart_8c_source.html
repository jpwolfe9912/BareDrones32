1093HAL_StatusTypeDef HAL_UART_Transmit(UART_HandleTypeDef *huart, const uint8_t *pData, uint16_t Size, uint32_t Timeout)
1094{
1095 const uint8_t *pdata8bits;
1096 const uint16_t *pdata16bits;
1097 uint32_t tickstart;
1098
1099 /* Check that a Tx process is not already ongoing */
1100 if (huart->gState == HAL_UART_STATE_READY)
1101 {
1102 if ((pData == NULL) || (Size == 0U))
1103 {
1104 return HAL_ERROR;
1105 }
1106
1107 __HAL_LOCK(huart);
1108
1109 huart->ErrorCode = HAL_UART_ERROR_NONE;
1110 huart->gState = HAL_UART_STATE_BUSY_TX;
1111
1112 /* Init tickstart for timeout management */
1113 tickstart = HAL_GetTick();
1114
1115 huart->TxXferSize = Size;
1116 huart->TxXferCount = Size;
1117
1118 /* In case of 9bits/No Parity transfer, pData needs to be handled as a uint16_t pointer */
1119 if ((huart->Init.WordLength == UART_WORDLENGTH_9B) && (huart->Init.Parity == UART_PARITY_NONE))
1120 {
1121 pdata8bits = NULL;
1122 pdata16bits = (const uint16_t *) pData;
1123 }
1124 else
1125 {
1126 pdata8bits = pData;
1127 pdata16bits = NULL;
1128 }
1129
1130 __HAL_UNLOCK(huart);
1131
1132 while (huart->TxXferCount > 0U)
1133 {
1134 if (UART_WaitOnFlagUntilTimeout(huart, UART_FLAG_TXE, RESET, tickstart, Timeout) != HAL_OK)
1135 {
1136 return HAL_TIMEOUT;
1137 }
1138 if (pdata8bits == NULL)
1139 {
1140 huart->Instance->TDR = (uint16_t)(*pdata16bits & 0x01FFU);
1141 pdata16bits++;
1142 }
1143 else
1144 {
1145 huart->Instance->TDR = (uint8_t)(*pdata8bits & 0xFFU);
1146 pdata8bits++;
1147 }
1148 huart->TxXferCount--;
1149 }
1150
1151 if (UART_WaitOnFlagUntilTimeout(huart, UART_FLAG_TC, RESET, tickstart, Timeout) != HAL_OK)
1152 {
1153 return HAL_TIMEOUT;
1154 }
1155
1156 /* At end of Tx process, restore huart->gState to Ready */
1157 huart->gState = HAL_UART_STATE_READY;
1158
1159 return HAL_OK;
1160 }
1161 else
1162 {
1163 return HAL_BUSY;
1164 }

1182 uint16_t uhMask;
1183 uint32_t tickstart;
1184
1185 /* Check that a Rx process is not already ongoing */
1186 if (huart->RxState == HAL_UART_STATE_READY)
1187 {
1188 if ((pData == NULL) || (Size == 0U))
1189 {
1190 return HAL_ERROR;
1191 }
1192
1193 __HAL_LOCK(huart);
1194
1195 huart->ErrorCode = HAL_UART_ERROR_NONE;
1196 huart->RxState = HAL_UART_STATE_BUSY_RX;
1197 huart->ReceptionType = HAL_UART_RECEPTION_STANDARD;
1198
1199 /* Init tickstart for timeout management */
1200 tickstart = HAL_GetTick();
1201
1202 huart->RxXferSize = Size;
1203 huart->RxXferCount = Size;
1204
1205 /* Computation of UART mask to apply to RDR register */
1206 UART_MASK_COMPUTATION(huart);
1207 uhMask = huart->Mask;
1208
1209 /* In case of 9bits/No Parity transfer, pRxData needs to be handled as a uint16_t pointer */
1210 if ((huart->Init.WordLength == UART_WORDLENGTH_9B) && (huart->Init.Parity == UART_PARITY_NONE))
1211 {
1212 pdata8bits = NULL;
1213 pdata16bits = (uint16_t *) pData;
1214 }
1215 else
1216 {
1217 pdata8bits = pData;
1218 pdata16bits = NULL;
1219 }
1220
1221 __HAL_UNLOCK(huart);
1222
1223 /* as long as data have to be received */

1225 {

1227 {
1228 return HAL_TIMEOUT;
1229 }
1230 if (pdata8bits == NULL)
1231 {
1232 *pdata16bits = (uint16_t)(huart->Instance->RDR & uhMask);
1233 pdata16bits++;
1234 }
1235 else
1236 {
1237 *pdata8bits = (uint8_t)(huart->Instance->RDR & (uint8_t)uhMask);
1238 pdata8bits++;
1239 }
1240 huart->RxXferCount--;
1241 }
1242
1243 /* At end of Rx process, restore huart->RxState to Ready */
1244 huart->RxState = HAL_UART_STATE_READY;
1245
1246 return HAL_OK;
1247 }
1248 else
1249 {
1250 return HAL_BUSY;
1251 }

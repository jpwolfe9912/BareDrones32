62#define LANE4567_MASK 0xFF00

64void arm_levinson_durbin_f16(const float16_t *phi,
65 float16_t *a,
66 float16_t *err,
67 int nbCoefs)
68{
69 _Float16 e;
70 static const uint16_t revOffsetArray[8] = {7,6,5,4,3,2,1,0};
71
72 a[0] = (_Float16)phi[1] / (_Float16)phi[0];
73
74 e = (_Float16)phi[0] - (_Float16)phi[1] * (_Float16)a[0];
75 for(int p=1; p < nbCoefs; p++)
76 {
77 _Float16 suma = 0.0f16;
78 _Float16 sumb = 0.0f16;
79 f16x8_t vecA,vecRevPhi,vecPhi,vecSumA, vecSumB;
80 _Float16 k;
81 uint32_t blkCnt;
82 const float16_t *pPhi,*pRevPhi,*pA;
83 uint16x8_t revOffset;
84
85 int nb,j,i;
86
87 revOffset = vld1q(revOffsetArray);
88 vecSumA = vdupq_n_f16(0.0f16);
89 vecSumB = vdupq_n_f16(0.0f16);
90
91 pRevPhi = &phi[p-7];
92 pPhi = &phi[1];
93 pA = a;
94
95 i = 0;
96 blkCnt = p >> 3;
97 while(blkCnt > 0)
98 {
99 vecA = vld1q(pA);
100 pA += 8;
101
102 vecPhi = vld1q(pPhi);
103 pPhi += 8;
104
105 vecRevPhi = vldrhq_gather_shifted_offset_f16(pRevPhi,revOffset);
106 pRevPhi -= 8;
107
108 vecSumA = vfmaq(vecSumA,vecA,vecRevPhi);
109 vecSumB = vfmaq(vecSumB,vecA,vecPhi);
110
111 i += 8;
112 blkCnt--;
113
114 }
115
116 suma = vecAddAcrossF16Mve(vecSumA);
117 sumb = vecAddAcrossF16Mve(vecSumB);
118
119 blkCnt = p & 7;
120 while(blkCnt > 0)
121 {
122 suma += (_Float16)a[i] * (_Float16)phi[p - i];
123 sumb += (_Float16)a[i] * (_Float16)phi[i + 1];
124
125 i++;
126 blkCnt--;
127 }
128
129 k = ((_Float16)phi[p+1] - suma)/((_Float16)phi[0] - sumb);
130
131 f16x8_t vecRevA,tmp;
132 static uint16_t orgOffsetArray[8]={0,1,2,3,-1,-2,-3,-4};
133 static const uint16_t offsetIncArray[8]={4,4,4,4,-4,-4,-4,-4};
134
135 uint16x8_t offset,offsetInc,vecTmp;
136
137
138 offset = vld1q(orgOffsetArray);
139 vecTmp = vdupq_n_u16(p);
140
141 offset = vaddq_m_u16(offset,offset,vecTmp,LANE4567_MASK);
142 offsetInc = vld1q(offsetIncArray);
143
144 nb = p >> 3;
145 j=0;
146 for(int i = 0; i < nb ; i++)
147 {
148
149 /*
150 x0=a[j] - k * a[p-1-j];
151 x1=a[j+1] - k * a[p-2-j];
152 x3=a[p-1-j] - k * a[j];
153 x4=a[p-2-j] - k * a[j+1];
154
155 a[j] = x0;
156 a[j+1] = x1;
157 a[p-1-j] = x2;
158 a[p-2-j] = x3;
159 */
160
161 uint64_t tmpa,tmpb;
162 vecA = vldrhq_gather_shifted_offset_f16(a,offset);
163
164
165 tmpa = vgetq_lane_u64((uint64x2_t)vecA,0);
166 tmpb = vgetq_lane_u64((uint64x2_t)vecA,1);
167 vecRevA = (f16x8_t) vsetq_lane_u64(tmpb,(uint64x2_t)vecRevA,0);
168 vecRevA = (f16x8_t) vsetq_lane_u64(tmpa,(uint64x2_t)vecRevA,1);
169
170
171 tmp = vsubq(vecA,vmulq_n_f16(vecRevA,k));
172 vstrhq_scatter_shifted_offset_f16(a, offset, tmp);
173
174 offset = vaddq(offset,offsetInc);
175
176 j+=4;
177
178 }
179
180 blkCnt = p & 7;
181
182 if (blkCnt)
183 {
184 nb = blkCnt >> 1;
185 for(int i =0;i < nb ; i++)
186 {
187 _Float16 x,y;
188
189 x=(_Float16)a[j] - (_Float16)k * (_Float16)a[p-1-j];
190 y=(_Float16)a[p-1-j] - (_Float16)k * (_Float16)a[j];
191
192 a[j] = x;
193 a[p-1-j] = y;
194
195 j++;
196 }
197
198 nb = blkCnt & 1;
199 if (nb)
200 {
201 a[j]=(_Float16)a[j]- (_Float16)k * (_Float16)a[p-1-j];
202 }
203 }
204
205
206 a[p] = k;
207 e = e * (1.0f16 - k*k);
208
209
210 }
211 *err = e;
212}

218void arm_levinson_durbin_f16(const float16_t *phi,
219 float16_t *a,
220 float16_t *err,
221 int nbCoefs)
222{
223 _Float16 e;
224
225 a[0] = (_Float16)phi[1] / (_Float16)phi[0];
226
227 e = (_Float16)phi[0] - (_Float16)phi[1] * (_Float16)a[0];
228 for(int p=1; p < nbCoefs; p++)
229 {
230 _Float16 suma=0.0f16;
231 _Float16 sumb=0.0f16;
232 _Float16 k;
233 int nb,j;
234
235 for(int i=0; i < p; i++)
236 {
237 suma += (_Float16)a[i] * (_Float16)phi[p - i];
238 sumb += (_Float16)a[i] * (_Float16)phi[i + 1];
239 }
240
241 k = ((_Float16)phi[p+1]-suma)/((_Float16)phi[0] - sumb);
242
243
244 nb = p >> 1;
245 j=0;
246 for(int i =0;i < nb ; i++)
247 {
248 _Float16 x,y;
249
250 x=(_Float16)a[j] - (_Float16)k * (_Float16)a[p-1-j];
251 y=(_Float16)a[p-1-j] - (_Float16)k * (_Float16)a[j];
252
253 a[j] = x;
254 a[p-1-j] = y;
255
256 j++;
257 }
258
259 nb = p & 1;
260 if (nb)
261 {
262 a[j]=(_Float16)a[j]- (_Float16)k * (_Float16)a[p-1-j];
263 }
264
265 a[p] = k;
266 e = e * (1.0f16 - k*k);
267
268
269 }
270 *err = e;
271}

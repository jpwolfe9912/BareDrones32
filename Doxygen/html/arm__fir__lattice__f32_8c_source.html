121void arm_fir_lattice_f32(

123 const float32_t * pSrc,
124 float32_t * pDst,
125 uint32_t blockSize)
126{
127 float32_t *pState = S->pState; /* State pointer */
128 const float32_t *pCoeffs = S->pCoeffs; /* Coefficient pointer */
129 float32_t *px; /* Temporary state pointer */
130 const float32_t *pk; /* Temporary coefficient pointer */
131 uint32_t numStages = S->numStages; /* Number of stages in the filter */
132 uint32_t blkCnt, stageCnt; /* Loop counters */
133 float32_t fcurr0, fnext0, gnext0, gcurr0; /* Temporary variables */
134
135#if defined (ARM_MATH_LOOPUNROLL)
136 float32_t fcurr1, fnext1, gnext1; /* Temporary variables for second sample in loop unrolling */
137 float32_t fcurr2, fnext2, gnext2; /* Temporary variables for third sample in loop unrolling */
138 float32_t fcurr3, fnext3, gnext3; /* Temporary variables for fourth sample in loop unrolling */
139#endif
140
141 gcurr0 = 0.0f;
142
143#if defined (ARM_MATH_LOOPUNROLL)
144
145 /* Loop unrolling: Compute 4 outputs at a time */
146 blkCnt = blockSize >> 2U;
147
148 while (blkCnt > 0U)
149 {
150 /* Read two samples from input buffer */
151 /* f0(n) = x(n) */
152 fcurr0 = *pSrc++;
153 fcurr1 = *pSrc++;
154
155 /* Initialize state pointer */
156 px = pState;
157
158 /* Initialize coeff pointer */
159 pk = pCoeffs;
160
161 /* Read g0(n-1) from state buffer */
162 gcurr0 = *px;
163
164 /* Process first sample for first tap */
165 /* f1(n) = f0(n) + K1 * g0(n-1) */
166 fnext0 = (gcurr0 * (*pk)) + fcurr0;
167
168 /* g1(n) = f0(n) * K1 + g0(n-1) */
169 gnext0 = (fcurr0 * (*pk)) + gcurr0;
170
171 /* Process second sample for first tap */
172 fnext1 = (fcurr0 * (*pk)) + fcurr1;
173 gnext1 = (fcurr1 * (*pk)) + fcurr0;
174
175 /* Read next two samples from input buffer */
176 /* f0(n+2) = x(n+2) */
177 fcurr2 = *pSrc++;
178 fcurr3 = *pSrc++;
179
180 /* Process third sample for first tap */
181 fnext2 = (fcurr1 * (*pk)) + fcurr2;
182 gnext2 = (fcurr2 * (*pk)) + fcurr1;
183
184 /* Process fourth sample for first tap */
185 fnext3 = (fcurr2 * (*pk )) + fcurr3;
186 gnext3 = (fcurr3 * (*pk++)) + fcurr2;
187
188 /* Copy only last input sample into the state buffer
189 which will be used for next samples processing */
190 *px++ = fcurr3;
191
192 /* Update of f values for next coefficient set processing */
193 fcurr0 = fnext0;
194 fcurr1 = fnext1;
195 fcurr2 = fnext2;
196 fcurr3 = fnext3;
197
198 /* Loop unrolling. Process 4 taps at a time . */
199 stageCnt = (numStages - 1U) >> 2U;
200
201 /* Loop over the number of taps. Unroll by a factor of 4.
202 Repeat until we've computed numStages-3 coefficients. */
203
204 /* Process 2nd, 3rd, 4th and 5th taps ... here */
205 while (stageCnt > 0U)
206 {
207 /* Read g1(n-1), g3(n-1) .... from state */
208 gcurr0 = *px;
209
210 /* save g1(n) in state buffer */
211 *px++ = gnext3;
212
213 /* Process first sample for 2nd, 6th .. tap */
214 /* Sample processing for K2, K6.... */
215 /* f2(n) = f1(n) + K2 * g1(n-1) */
216 fnext0 = (gcurr0 * (*pk)) + fcurr0;
217
218 /* Process second sample for 2nd, 6th .. tap */
219 /* for sample 2 processing */
220 fnext1 = (gnext0 * (*pk)) + fcurr1;
221
222 /* Process third sample for 2nd, 6th .. tap */
223 fnext2 = (gnext1 * (*pk)) + fcurr2;
224
225 /* Process fourth sample for 2nd, 6th .. tap */
226 fnext3 = (gnext2 * (*pk)) + fcurr3;
227
228 /* g2(n) = f1(n) * K2 + g1(n-1) */
229 /* Calculation of state values for next stage */
230 gnext3 = (fcurr3 * (*pk)) + gnext2;
231
232 gnext2 = (fcurr2 * (*pk)) + gnext1;
233
234 gnext1 = (fcurr1 * (*pk)) + gnext0;
235
236 gnext0 = (fcurr0 * (*pk++)) + gcurr0;
237
238
239 /* Read g2(n-1), g4(n-1) .... from state */
240 gcurr0 = *px;
241
242 /* save g2(n) in state buffer */
243 *px++ = gnext3;
244
245 /* Sample processing for K3, K7.... */
246 /* Process first sample for 3rd, 7th .. tap */
247 /* f3(n) = f2(n) + K3 * g2(n-1) */
248 fcurr0 = (gcurr0 * (*pk)) + fnext0;
249
250 /* Process second sample for 3rd, 7th .. tap */
251 fcurr1 = (gnext0 * (*pk)) + fnext1;
252
253 /* Process third sample for 3rd, 7th .. tap */
254 fcurr2 = (gnext1 * (*pk)) + fnext2;
255
256 /* Process fourth sample for 3rd, 7th .. tap */
257 fcurr3 = (gnext2 * (*pk)) + fnext3;
258
259 /* Calculation of state values for next stage */
260 /* g3(n) = f2(n) * K3 + g2(n-1) */
261 gnext3 = (fnext3 * (*pk)) + gnext2;
262
263 gnext2 = (fnext2 * (*pk)) + gnext1;
264
265 gnext1 = (fnext1 * (*pk)) + gnext0;
266
267 gnext0 = (fnext0 * (*pk++)) + gcurr0;
268
269
270 /* Read g1(n-1), g3(n-1) .... from state */
271 gcurr0 = *px;
272
273 /* save g3(n) in state buffer */
274 *px++ = gnext3;
275
276 /* Sample processing for K4, K8.... */
277 /* Process first sample for 4th, 8th .. tap */
278 /* f4(n) = f3(n) + K4 * g3(n-1) */
279 fnext0 = (gcurr0 * (*pk)) + fcurr0;
280
281 /* Process second sample for 4th, 8th .. tap */
282 /* for sample 2 processing */
283 fnext1 = (gnext0 * (*pk)) + fcurr1;
284
285 /* Process third sample for 4th, 8th .. tap */
286 fnext2 = (gnext1 * (*pk)) + fcurr2;
287
288 /* Process fourth sample for 4th, 8th .. tap */
289 fnext3 = (gnext2 * (*pk)) + fcurr3;
290
291 /* g4(n) = f3(n) * K4 + g3(n-1) */
292 /* Calculation of state values for next stage */
293 gnext3 = (fcurr3 * (*pk)) + gnext2;
294
295 gnext2 = (fcurr2 * (*pk)) + gnext1;
296
297 gnext1 = (fcurr1 * (*pk)) + gnext0;
298
299 gnext0 = (fcurr0 * (*pk++)) + gcurr0;
300
301
302 /* Read g2(n-1), g4(n-1) .... from state */
303 gcurr0 = *px;
304
305 /* save g4(n) in state buffer */
306 *px++ = gnext3;
307
308 /* Sample processing for K5, K9.... */
309 /* Process first sample for 5th, 9th .. tap */
310 /* f5(n) = f4(n) + K5 * g4(n-1) */
311 fcurr0 = (gcurr0 * (*pk)) + fnext0;
312
313 /* Process second sample for 5th, 9th .. tap */
314 fcurr1 = (gnext0 * (*pk)) + fnext1;
315
316 /* Process third sample for 5th, 9th .. tap */
317 fcurr2 = (gnext1 * (*pk)) + fnext2;
318
319 /* Process fourth sample for 5th, 9th .. tap */
320 fcurr3 = (gnext2 * (*pk)) + fnext3;
321
322 /* Calculation of state values for next stage */
323 /* g5(n) = f4(n) * K5 + g4(n-1) */
324 gnext3 = (fnext3 * (*pk)) + gnext2;
325
326 gnext2 = (fnext2 * (*pk)) + gnext1;
327
328 gnext1 = (fnext1 * (*pk)) + gnext0;
329
330 gnext0 = (fnext0 * (*pk++)) + gcurr0;
331
332 stageCnt--;
333 }
334
335 /* If the (filter length -1) is not a multiple of 4, compute the remaining filter taps */
336 stageCnt = (numStages - 1U) % 0x4U;
337
338 while (stageCnt > 0U)
339 {
340 gcurr0 = *px;
341
342 /* save g value in state buffer */
343 *px++ = gnext3;
344
345 /* Process four samples for last three taps here */
346 fnext0 = (gcurr0 * (*pk)) + fcurr0;
347
348 fnext1 = (gnext0 * (*pk)) + fcurr1;
349
350 fnext2 = (gnext1 * (*pk)) + fcurr2;
351
352 fnext3 = (gnext2 * (*pk)) + fcurr3;
353
354 /* g1(n) = f0(n) * K1 + g0(n-1) */
355 gnext3 = (fcurr3 * (*pk)) + gnext2;
356
357 gnext2 = (fcurr2 * (*pk)) + gnext1;
358
359 gnext1 = (fcurr1 * (*pk)) + gnext0;
360
361 gnext0 = (fcurr0 * (*pk++)) + gcurr0;
362
363 /* Update of f values for next coefficient set processing */
364 fcurr0 = fnext0;
365 fcurr1 = fnext1;
366 fcurr2 = fnext2;
367 fcurr3 = fnext3;
368
369 stageCnt--;
370 }
371
372 /* The results in the 4 accumulators, store in the destination buffer. */
373 /* y(n) = fN(n) */
374 *pDst++ = fcurr0;
375 *pDst++ = fcurr1;
376 *pDst++ = fcurr2;
377 *pDst++ = fcurr3;
378
379 blkCnt--;
380 }
381
382 /* Loop unrolling: Compute remaining outputs */
383 blkCnt = blockSize % 0x4U;
384
385#else
386
387 /* Initialize blkCnt with number of samples */
388 blkCnt = blockSize;
389
390#endif /* #if defined (ARM_MATH_LOOPUNROLL) */
391
392 while (blkCnt > 0U)
393 {
394 /* f0(n) = x(n) */
395 fcurr0 = *pSrc++;
396
397 /* Initialize state pointer */
398 px = pState;
399
400 /* Initialize coeff pointer */
401 pk = pCoeffs;
402
403 /* read g2(n) from state buffer */
404 gcurr0 = *px;
405
406 /* for sample 1 processing */
407 /* f1(n) = f0(n) + K1 * g0(n-1) */
408 fnext0 = (gcurr0 * (*pk)) + fcurr0;
409
410 /* g1(n) = f0(n) * K1 + g0(n-1) */
411 gnext0 = (fcurr0 * (*pk++)) + gcurr0;
412
413 /* save g1(n) in state buffer */
414 *px++ = fcurr0;
415
416 /* f1(n) is saved in fcurr0 for next stage processing */
417 fcurr0 = fnext0;
418
419 stageCnt = (numStages - 1U);
420
421 /* stage loop */
422 while (stageCnt > 0U)
423 {
424 /* read g2(n) from state buffer */
425 gcurr0 = *px;
426
427 /* save g1(n) in state buffer */
428 *px++ = gnext0;
429
430 /* Sample processing for K2, K3.... */
431 /* f2(n) = f1(n) + K2 * g1(n-1) */
432 fnext0 = (gcurr0 * (*pk)) + fcurr0;
433
434 /* g2(n) = f1(n) * K2 + g1(n-1) */
435 gnext0 = (fcurr0 * (*pk++)) + gcurr0;
436
437 /* f1(n) is saved in fcurr0 for next stage processing */
438 fcurr0 = fnext0;
439
440 stageCnt--;
441 }
442
443 /* y(n) = fN(n) */

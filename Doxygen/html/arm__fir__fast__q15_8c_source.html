60void arm_fir_fast_q15(
61 const arm_fir_instance_q15 * S,
62 const q15_t * pSrc,
63 q15_t * pDst,
64 uint32_t blockSize)
65{
66 q15_t *pState = S->pState; /* State pointer */
67 const q15_t *pCoeffs = S->pCoeffs; /* Coefficient pointer */
68 q15_t *pStateCurnt; /* Points to the current sample of the state */
69 q15_t *px; /* Temporary pointer for state buffer */
70 const q15_t *pb; /* Temporary pointer for coefficient buffer */
71 q31_t acc0; /* Accumulators */
72 uint32_t numTaps = S->numTaps; /* Number of filter coefficients in the filter */
73 uint32_t tapCnt, blkCnt; /* Loop counters */
74
75#if defined (ARM_MATH_LOOPUNROLL)
76 q31_t acc1, acc2, acc3; /* Accumulators */
77 q31_t x0, x1, x2, c0; /* Temporary variables to hold state and coefficient values */
78#endif
79
80 /* S->pState points to state array which contains previous frame (numTaps - 1) samples */
81 /* pStateCurnt points to the location where the new input data should be written */
82 pStateCurnt = &(S->pState[(numTaps - 1U)]);
83
84#if defined (ARM_MATH_LOOPUNROLL)
85
86 /* Loop unrolling: Compute 4 output values simultaneously.
87 * The variables acc0 ... acc3 hold output values that are being computed:
88 *
89 * acc0 = b[numTaps-1] * x[n-numTaps-1] + b[numTaps-2] * x[n-numTaps-2] + b[numTaps-3] * x[n-numTaps-3] +...+ b[0] * x[0]
90 * acc1 = b[numTaps-1] * x[n-numTaps] + b[numTaps-2] * x[n-numTaps-1] + b[numTaps-3] * x[n-numTaps-2] +...+ b[0] * x[1]
91 * acc2 = b[numTaps-1] * x[n-numTaps+1] + b[numTaps-2] * x[n-numTaps] + b[numTaps-3] * x[n-numTaps-1] +...+ b[0] * x[2]
92 * acc3 = b[numTaps-1] * x[n-numTaps+2] + b[numTaps-2] * x[n-numTaps+1] + b[numTaps-3] * x[n-numTaps] +...+ b[0] * x[3]
93 */
94 blkCnt = blockSize >> 2U;
95
96 while (blkCnt > 0U)
97 {
98 /* Copy 4 new input samples into the state buffer. */
99 *pStateCurnt++ = *pSrc++;
100 *pStateCurnt++ = *pSrc++;
101 *pStateCurnt++ = *pSrc++;
102 *pStateCurnt++ = *pSrc++;
103
104 /* Set all accumulators to zero */
105 acc0 = 0;
106 acc1 = 0;
107 acc2 = 0;
108 acc3 = 0;
109
110 /* Typecast q15_t pointer to q31_t pointer for state reading in q31_t */
111 px = pState;
112
113 /* Typecast q15_t pointer to q31_t pointer for coefficient reading in q31_t */
114 pb = pCoeffs;
115
116 /* Read the first two samples from the state buffer: x[n-N], x[n-N-1] */
117 x0 = read_q15x2_ia (&px);
118
119 /* Read the third and forth samples from the state buffer: x[n-N-2], x[n-N-3] */
120 x2 = read_q15x2_ia (&px);
121
122 /* Loop over the number of taps. Unroll by a factor of 4.
123 Repeat until we've computed numTaps-(numTaps%4) coefficients. */
124 tapCnt = numTaps >> 2U;
125
126 while (tapCnt > 0U)
127 {
128 /* Read the first two coefficients using SIMD: b[N] and b[N-1] coefficients */
129 c0 = read_q15x2_ia ((q15_t **) &pb);
130
131 /* acc0 += b[N] * x[n-N] + b[N-1] * x[n-N-1] */
132 acc0 = __SMLAD(x0, c0, acc0);
133
134 /* acc2 += b[N] * x[n-N-2] + b[N-1] * x[n-N-3] */
135 acc2 = __SMLAD(x2, c0, acc2);
136
137 /* pack x[n-N-1] and x[n-N-2] */
138#ifndef ARM_MATH_BIG_ENDIAN
139 x1 = __PKHBT(x2, x0, 0);
140#else
141 x1 = __PKHBT(x0, x2, 0);
142#endif
143
144 /* Read state x[n-N-4], x[n-N-5] */
145 x0 = read_q15x2_ia (&px);
146
147 /* acc1 += b[N] * x[n-N-1] + b[N-1] * x[n-N-2] */
148 acc1 = __SMLADX(x1, c0, acc1);
149
150 /* pack x[n-N-3] and x[n-N-4] */
151#ifndef ARM_MATH_BIG_ENDIAN
152 x1 = __PKHBT(x0, x2, 0);
153#else
154 x1 = __PKHBT(x2, x0, 0);
155#endif
156
157 /* acc3 += b[N] * x[n-N-3] + b[N-1] * x[n-N-4] */
158 acc3 = __SMLADX(x1, c0, acc3);
159
160 /* Read coefficients b[N-2], b[N-3] */
161 c0 = read_q15x2_ia ((q15_t **) &pb);
162
163 /* acc0 += b[N-2] * x[n-N-2] + b[N-3] * x[n-N-3] */
164 acc0 = __SMLAD(x2, c0, acc0);
165
166 /* Read state x[n-N-6], x[n-N-7] with offset */
167 x2 = read_q15x2_ia (&px);
168
169 /* acc2 += b[N-2] * x[n-N-4] + b[N-3] * x[n-N-5] */
170 acc2 = __SMLAD(x0, c0, acc2);
171
172 /* acc1 += b[N-2] * x[n-N-3] + b[N-3] * x[n-N-4] */
173 acc1 = __SMLADX(x1, c0, acc1);
174
175 /* pack x[n-N-5] and x[n-N-6] */
176#ifndef ARM_MATH_BIG_ENDIAN
177 x1 = __PKHBT(x2, x0, 0);
178#else
179 x1 = __PKHBT(x0, x2, 0);
180#endif
181
182 /* acc3 += b[N-2] * x[n-N-5] + b[N-3] * x[n-N-6] */
183 acc3 = __SMLADX(x1, c0, acc3);
184
185 /* Decrement tap count */
186 tapCnt--;
187 }
188
189 /* If the filter length is not a multiple of 4, compute the remaining filter taps.
190 This is always be 2 taps since the filter length is even. */
191 if ((numTaps & 0x3U) != 0U)
192 {
193 /* Read last two coefficients */
194 c0 = read_q15x2_ia ((q15_t **) &pb);
195
196 /* Perform the multiply-accumulates */
197 acc0 = __SMLAD(x0, c0, acc0);
198 acc2 = __SMLAD(x2, c0, acc2);
199
200 /* pack state variables */
201#ifndef ARM_MATH_BIG_ENDIAN
202 x1 = __PKHBT(x2, x0, 0);
203#else
204 x1 = __PKHBT(x0, x2, 0);
205#endif
206
207 /* Read last state variables */
208 x0 = read_q15x2 (px);
209
210 /* Perform the multiply-accumulates */
211 acc1 = __SMLADX(x1, c0, acc1);
212
213 /* pack state variables */
214#ifndef ARM_MATH_BIG_ENDIAN
215 x1 = __PKHBT(x0, x2, 0);
216#else
217 x1 = __PKHBT(x2, x0, 0);
218#endif
219
220 /* Perform the multiply-accumulates */
221 acc3 = __SMLADX(x1, c0, acc3);
222 }
223
224 /* The results in the 4 accumulators are in 2.30 format. Convert to 1.15 with saturation.
225 Then store the 4 outputs in the destination buffer. */
226#ifndef ARM_MATH_BIG_ENDIAN
227 write_q15x2_ia (&pDst, __PKHBT(__SSAT((acc0 >> 15), 16), __SSAT((acc1 >> 15), 16), 16));
228 write_q15x2_ia (&pDst, __PKHBT(__SSAT((acc2 >> 15), 16), __SSAT((acc3 >> 15), 16), 16));
229#else
230 write_q15x2_ia (&pDst, __PKHBT(__SSAT((acc1 >> 15), 16), __SSAT((acc0 >> 15), 16), 16));
231 write_q15x2_ia (&pDst, __PKHBT(__SSAT((acc3 >> 15), 16), __SSAT((acc2 >> 15), 16), 16));
232#endif /* #ifndef ARM_MATH_BIG_ENDIAN */
233
234 /* Advance the state pointer by 4 to process the next group of 4 samples */
235 pState = pState + 4U;
236
237 /* Decrement loop counter */
238 blkCnt--;
239 }
240
241 /* Loop unrolling: Compute remaining output samples */
242 blkCnt = blockSize % 0x4U;
243
244#else
245
246 /* Initialize blkCnt with number of taps */
247 blkCnt = blockSize;
248
249#endif /* #if defined (ARM_MATH_LOOPUNROLL) */
250
251 while (blkCnt > 0U)
252 {
253 /* Copy two samples into state buffer */
254 *pStateCurnt++ = *pSrc++;
255
256 /* Set the accumulator to zero */
257 acc0 = 0;
258
259 /* Use SIMD to hold states and coefficients */
260 px = pState;
261 pb = pCoeffs;
262
263 tapCnt = numTaps >> 1U;
264
265 do
266 {
267 acc0 += (q31_t) *px++ * *pb++;
268 acc0 += (q31_t) *px++ * *pb++;
269
270 tapCnt--;
271 }
272 while (tapCnt > 0U);
273
274 /* The result is in 2.30 format. Convert to 1.15 with saturation.
275 Then store the output in the destination buffer. */
276 *pDst++ = (q15_t) (__SSAT((acc0 >> 15), 16));
277
278 /* Advance state pointer by 1 for the next sample */
279 pState = pState + 1U;
280
281 /* Decrement loop counter */
282 blkCnt--;
283 }
284
285 /* Processing is complete.
286 Now copy the last numTaps - 1 samples to the start of the state buffer.
287 This prepares the state buffer for the next function call. */
288
289 /* Points to the start of the state buffer */
290 pStateCurnt = S->pState;
291
292#if defined (ARM_MATH_LOOPUNROLL)
293
294 /* Loop unrolling: Compute 4 taps at a time */
295 tapCnt = (numTaps - 1U) >> 2U;
296
297 /* Copy data */
298 while (tapCnt > 0U)
299 {
300 *pStateCurnt++ = *pState++;
301 *pStateCurnt++ = *pState++;
302 *pStateCurnt++ = *pState++;
303 *pStateCurnt++ = *pState++;
304
305 /* Decrement loop counter */
306 tapCnt--;
307 }
308
309 /* Calculate remaining number of copies */
310 tapCnt = (numTaps - 1U) % 0x4U;
311
312#else
313
314 /* Initialize tapCnt with number of taps */
315 tapCnt = (numTaps - 1U);
316
317#endif /* #if defined (ARM_MATH_LOOPUNROLL) */
318
319 /* Copy remaining data */
320 while (tapCnt > 0U)
321 {
322 *pStateCurnt++ = *pState++;
323
324 /* Decrement loop counter */
325 tapCnt--;
326 }
327
328}
